## Execution Script

### Full Cleanup Script (PowerShell)

```powershell
# PROJECT_CLEANUP.ps1
# Run from project root: C:\cursor\HomeIQ

Set-Location "C:\cursor\HomeIQ"

Write-Host "=== Phase 1.1: Deleting temporary Python scripts ===" -ForegroundColor Cyan
$pythonScripts = @(
    "analyze_pattern_synergy_data.py",
    "check_weather_data.py",
    "cleanup_patterns_synergies.py",
    "direct_store_docs.py",
    "get_validation_errors.py",
    "populate_all_docs.py",
    "populate_cache_with_mcp.py",
    "populate_context7_cache.py",
    "populate_with_api_key.py",
    "quick_analysis.py",
    "store_api_key.py",
    "store_context7_docs.py",
    "test_cache_lock.py",
    "test_self_correction.py",
    "verify_context7.py"
)
foreach ($file in $pythonScripts) {
    if (Test-Path $file) {
        Remove-Item $file -Force
        Write-Host "  Deleted: $file" -ForegroundColor Green
    }
}

Write-Host "`n=== Phase 1.2: Deleting debug JSON files ===" -ForegroundColor Cyan
Get-ChildItem -Filter "debug_prompt_*.json" | Remove-Item -Force
$jsonFiles = @(
    "debug_analysis_output.json",
    "conversation-analysis.json",
    "coverage.json",
    "deployment_test_results.json",
    "test_result.json"
)
foreach ($file in $jsonFiles) {
    if (Test-Path $file) {
        Remove-Item $file -Force
        Write-Host "  Deleted: $file" -ForegroundColor Green
    }
}

Write-Host "`n=== Phase 1.3: Deleting temporary text files ===" -ForegroundColor Cyan
$txtFiles = @(
    "analysis_results.txt",
    "cleanup_dry_run.txt",
    "review-results.txt",
    "test_output.txt"
)
foreach ($file in $txtFiles) {
    if (Test-Path $file) {
        Remove-Item $file -Force
        Write-Host "  Deleted: $file" -ForegroundColor Green
    }
}

Write-Host "`n=== Phase 1.4: Deleting log files ===" -ForegroundColor Cyan
$logFiles = @(
    "build-output.log",
    "build-output-final.log",
    "quality_evaluation.log"
)
foreach ($file in $logFiles) {
    if (Test-Path $file) {
        Remove-Item $file -Force
        Write-Host "  Deleted: $file" -ForegroundColor Green
    }
}
if (Test-Path "ha_events.log") {
    Remove-Item "ha_events.log" -Recurse -Force
    Write-Host "  Deleted: ha_events.log/" -ForegroundColor Green
}

Write-Host "`n=== Phase 1.5: Moving misplaced markdown files ===" -ForegroundColor Cyan
if (Test-Path "DEPLOY_YAML_VALIDATION.md") {
    Move-Item "DEPLOY_YAML_VALIDATION.md" "implementation/" -Force
    Write-Host "  Moved: DEPLOY_YAML_VALIDATION.md -> implementation/" -ForegroundColor Green
}
if (Test-Path "PATTERN_SYNERGY_CLEANUP_PLAN.md") {
    Move-Item "PATTERN_SYNERGY_CLEANUP_PLAN.md" "implementation/" -Force
    Write-Host "  Moved: PATTERN_SYNERGY_CLEANUP_PLAN.md -> implementation/" -ForegroundColor Green
}
if (Test-Path "requirements.md") {
    Move-Item "requirements.md" "docs/" -Force
    Write-Host "  Moved: requirements.md -> docs/" -ForegroundColor Green
}
if (Test-Path "enhanced-prompt.md") {
    Remove-Item "enhanced-prompt.md" -Force
    Write-Host "  Deleted: enhanced-prompt.md" -ForegroundColor Green
}

Write-Host "`n=== Phase 2: Deleting empty directories ===" -ForegroundColor Cyan
$emptyDirs = @(
    "backups",
    "issues",
    "stories",
    "docs/workflows/simple-mode/archive"
)
foreach ($dir in $emptyDirs) {
    if (Test-Path $dir) {
        Remove-Item $dir -Recurse -Force -ErrorAction SilentlyContinue
        Write-Host "  Deleted: $dir/" -ForegroundColor Green
    }
}

Write-Host "`n=== Phase 4: Deleting old backup files ===" -ForegroundColor Cyan
if (Test-Path "ai_automation_backup.db") {
    Remove-Item "ai_automation_backup.db" -Force
    Write-Host "  Deleted: ai_automation_backup.db" -ForegroundColor Green
}
if (Test-Path "coverage.xml") {
    Remove-Item "coverage.xml" -Force
    Write-Host "  Deleted: coverage.xml" -ForegroundColor Green
}

Write-Host "`n=== Cleanup Complete! ===" -ForegroundColor Green
Write-Host "Run 'git status' to review changes before committing." -ForegroundColor Yellow
```
